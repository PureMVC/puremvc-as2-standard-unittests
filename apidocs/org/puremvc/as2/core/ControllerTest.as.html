     6  import org.puremvc.as2.core.*;
     7  import org.puremvc.as2.interfaces.IController;
     8  import org.puremvc.as2.patterns.command.MockSimpleCommand;
     9  import org.puremvc.as2.patterns.observer.Notification;
    10  import org.puremvc.as2.patterns.proxy.MockVO;
    11  
    12  class org.puremvc.as2.core.ControllerTest extends com.asunit.framework.TestCase {
    13  
    14  	private var className:String = "org.puremvc.as2.core.ControllerTest";
    15  
    16  	/**
    17  	 * Tests the Controller Singleton Factory Method 
    18  	 */
    19  	public function testGetInstance():Void {
    20  		
    21  		// Test Factory Method
    22  		var controller:IController = Controller.getInstance();
    23  		
    24  		// test assertions
    25  		assertTrue( "Expecting instance not null", controller != null );
    26  		assertTrue( "Expecting instance implements IController", controller instanceof IController);
    27  		
    28  		var secondController:IController = Controller.getInstance();
    29  		assertSame("two controller variable should point to the same object", controller, secondController);
    30  	}
    31  
    32  	/**
    33  	 * Tests Command registration and execution.
    34  	 * 
    35  	 * This test gets the Singleton Controller instance 
    36  	 * and registers the MockCommand class 
    37  	 * to handle 'ControllerTest' Notifications.<P>
    38  	 * 
    39  	 * It then constructs such a Notification and tells the 
    40  	 * Controller to execute the associated Command.
    41  	 * Success is determined by evaluating a property
    42  	 * on an object passed to the Command, which will
    43  	 * be modified when the Command executes.</P>
    44  	 * 
    45  	 */
    46  	public function testRegisterAndExecuteCommand():Void {
    47  		
    48  		// Create the controller, register the MockCommand to handle 'ControllerTest' notes
    49  		var controller:IController = Controller.getInstance();
    50  		controller.registerCommand("ControllerTest", MockSimpleCommand);
    51  		
    52  		// Create a 'ControllerTest' note
    53  		var vo:Object = new MockVO( 12 );
    54  		var note:Notification = new Notification( "ControllerTest", vo );
    55  
    56  		// Tell the controller to execute the Command associated with the note
    57  		// the ControllerTestCommand invoked will multiply the vo.input value
    58  		// by 2 and set the result on vo.result
    59  		controller.executeCommand(note);
    60  		
    61  		// test assertions 
    62  		assertTrue( "Expecting vo.result == 24", vo.result == 24 );
    63  	}
    64  	
    65  	/**
    66  	 * Tests Command registration and removal.
    67  	 * 
    68  	 * <P>
    69  	 * Tests that once a Command is registered and verified
    70  	 * working, it can be removed from the Controller.</P>
    71  	 */
    72  	public function testRegisterAndRemoveCommand():Void {
    73  		
    74  		// Create the controller, register the MockCommand to handle 'ControllerTest' notes
    75  		var controller:IController = Controller.getInstance();
    76  		controller.registerCommand("ControllerRemoveTest", MockSimpleCommand);
    77  		
    78  		// Create a 'ControllerTest' note
    79  		var vo:Object = new MockVO( 12 );
    80  		var note:Notification = new Notification( "ControllerRemoveTest", vo );
    81  
    82  		// Tell the controller to execute the Command associated with the note
    83  		// the ControllerTestCommand invoked will multiply the vo.input value
    84  		// by 2 and set the result on vo.result
    85  		controller.executeCommand(note);
    86  		
    87  		// test assertions 
    88  		assertTrue( "Expecting vo.result == 24", vo.result == 24 );
    89  		
    90  		// Reset result
    91  		vo.result = 0;
    92  		
    93  		// Remove the Command from the Controller
    94  		controller.removeCommand("ControllerRemoveTest");
    95  		
    96  		// Tell the controller to execute the Command associated with the
    97  		// note. This time, it should not be registered, and our vo result
    98  		// will not change   			
    99  		controller.executeCommand(note);
   100  		
   101  		// test assertions 
   102  		assertTrue( "Expecting vo.result == 0", vo.result == 0 );
   103  	}
   104    		
   105  	/**
   106  	 * Test hasCommand method.
   107  	 */
   108  	public function testHasCommand():Void {
   109  		
   110  		// register the MockCommand to handle 'hasCommandTest' notes
   111  		var controller:IController = Controller.getInstance();
   112  		controller.registerCommand("hasCommandTest", MockSimpleCommand);
   113  		
   114  		// test that hasCommand returns true for hasCommandTest notifications 
   115  		assertTrue( "Expecting controller.hasCommand('hasCommandTest') == true", controller.hasCommand("hasCommandTest") == true );
   116  		
   117  		// Remove the Command from the Controller
   118  		controller.removeCommand("hasCommandTest");
   119  		
   120  		// test that hasCommand returns false for hasCommandTest notifications 
   121  		assertTrue( "Expecting controller.hasCommand('hasCommandTest') == false", controller.hasCommand("hasCommandTest") == false );
   122  	}
   123  	
   124  }
