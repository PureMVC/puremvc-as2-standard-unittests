     6  import org.puremvc.as2.interfaces.*;
     7  
     8  /**
     9   * The interface definition for a PureMVC Mediator.
    10   *
    11   * <P>
    12   * In PureMVC, <code>IMediator</code> implementors assume these responsibilities:</P>
    13   * <UL>
    14   * <LI>Implement a common method which returns a list of all <code>INotification</code>s 
    15   * the <code>IMediator</code> has interest in.</LI>
    16   * <LI>Implement a common notification (callback) method.</LI>
    17   * </UL>
    18   * <P>
    19   * Additionally, <code>IMediator</code>s typically:
    20   * <UL>
    21   * <LI>Act as an intermediary between one or more view components such as text boxes or 
    22   * list controls, maintaining references and coordinating their behavior.</LI>
    23   * <LI>In Flash-based apps, this is often the place where event listeners are
    24   * added to view components, and their handlers implemented.</LI>
    25   * <LI>Respond to and generate <code>INotifications</code>, interacting with of 
    26   * the rest of the PureMVC app.
    27   * </UL></P>
    28   * <P>
    29   * When an <code>IMediator</code> is registered with the <code>IView</code>, 
    30   * the <code>IView</code> will call the <code>IMediator</code>'s 
    31   * <code>listNotificationInterests</code> method. The <code>IMediator</code> will 
    32   * return an <code>Array</code> of <code>INotification</code> names which 
    33   * it wishes to be notified about.</P>
    34   * 
    35   * <P>
    36   * The <code>IView</code> will then create an <code>Observer</code> object 
    37   * encapsulating that <code>IMediator</code>'s (<code>handleNotification</code>) method
    38   * and register it as an Observer for each <code>INotification</code> name returned by 
    39   * <code>listNotificationInterests</code>.</P>
    40   * 
    41   * <P>
    42   * A concrete IMediator implementor usually looks something like this:</P>
    43   * 
    44   * <listing>
    45   *	import org.puremvc.as2.patterns.mediator.~~;
    46   *	import org.puremvc.as2.patterns.observer.~~;
    47   *	import org.puremvc.as2.core.view.~~;
    48   * 
    49   *	import com.me.myapp.model.~~;
    50   *	import com.me.myapp.view.~~;
    51   *	import com.me.myapp.controller.~~;
    52   * 		
    53   *	import mx.controls.ComboBox;
    54   *	import mx.events.ListEvent;
    55   * 
    56   * public class MyMediator extends Mediator implements IMediator {
    57   * 
    58   * 		public function MyComboMediator( viewComponent:Object ) {
    59   * 			super( viewComponent );
    60   * 			combo.addEventListener( Event.CHANGE, onChange );
    61   * 		}
    62   * 		
    63   * 		public function listNotificationInterests():Array {
    64   * 				return [ MyFacade.SET_SELECTION, 
    65   * 						 MyFacade.SET_DATAPROVIDER ];
    66   * 		}
    67   * 
    68   * 		public function handleNotification( notification:INotification ):void {
    69   * 				switch ( notification.getName() ) {
    70   * 					case MyFacade.SET_SELECTION:
    71   * 						setSelection(notification);
    72   * 						break;
    73   * 					case MyFacade.SET_DATAPROVIDER:
    74   * 						setDataProvider(notification);
    75   * 						break;
    76   * 				}
    77   * 		}
    78   * 
    79   * 		// Set the data provider of the combo box
    80   * 		private function setDataProvider( notification:INotification ):void {
    81   * 			combo.dataProvider = notification.getBody() as Array;
    82   * 		}
    83   * 
    84   * 		// Invoked when the combo box dispatches a change event, we send a
    85   *      // notification with the
    86   * 		private function onChange(event:ListEvent):void {
    87   * 			sendNotification( MyFacade.MYCOMBO_CHANGED, this );
    88   * 		}
    89   * 
    90   * 		// A private getter for accessing the view object by class
    91   *      private function get combo():ComboBox  {
    92   *         return view as ComboBox;
    93   *      }
    94   * 
    95   * }
    96   * </listing>
    97   * 
    98   * @see org.puremvc.as2.interfaces.INotification INotification
    99   */
   100  interface org.puremvc.as2.interfaces.IMediator
   101  {
   102  	
   103  	/**
   104  	 * Get the <code>IMediator</code> instance name
   105  	 * 
   106  	 * @return the <code>IMediator</code> instance name
   107  	 */
   108  	function getMediatorName():String;
   109  	
   110  	/**
   111  	 * Get the <code>IMediator</code>'s view component.
   112  	 * 
   113  	 * @return Object the view component
   114  	 */
   115  	function getViewComponent():Object;
   116  	
   117  	/**
   118  	 * Set the <code>IMediator</code>'s view component.
   119  	 * 
   120  	 * @param Object the view component
   121  	 */
   122  	function setViewComponent( viewComponent:Object ):Void;
   123  		
   124  	/**
   125  	 * List <code>INotification</code> interests.
   126  	 * 
   127  	 * @return an <code>Array</code> of the <code>INotification</code> names this <code>IMediator</code> has an interest in.
   128  	 */
   129  	function listNotificationInterests( ):Array;
   130  	
   131  	/**
   132  	 * Handle an <code>INotification</code>.
   133  	 * 
   134  	 * @param notification the <code>INotification</code> to be handled
   135  	 */
   136  	function handleNotification( notification:INotification ):Void;
   137  	
   138  	/**
   139  	 * Called by the View when the Mediator is registered
   140  	 */ 
   141  	function onRegister( ):Void;
   142  
   143  	/**
   144  	 * Called by the View when the Mediator is removed
   145  	 */ 
   146  	function onRemove( ):Void;
   147  }
