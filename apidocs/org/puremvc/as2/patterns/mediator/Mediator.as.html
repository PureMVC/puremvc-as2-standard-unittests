     6  import org.puremvc.as2.interfaces.*;
     7  import org.puremvc.as2.patterns.observer.*;
     8  
     9  /**
    10   * A base <code>IMediator</code> implementation. 
    11   * 
    12   * @see org.puremvc.as2.core.view.View View
    13   */
    14  class org.puremvc.as2.patterns.mediator.Mediator extends Notifier implements IMediator, INotifier
    15  {
    16  
    17      /**
    18       * The name of the <code>Mediator</code>. 
    19       * 
    20       * <P>
    21       * Typically, a <code>Mediator</code> will be written to serve
    22       * one specific control or group controls and so,
    23  	 * will not have a need to be dynamically named.</P>
    24  	 */
    25  	public static var NAME : String = "Mediator";
    26  	
    27  	/**
    28  	 * Constructor.
    29  	 */
    30  	public function Mediator( mediatorName:String, viewComponent : Object) {
    31  		
    32  		this.mediatorName = (mediatorName != undefined)?mediatorName:NAME; 
    33  		this.viewComponent = viewComponent || null;	
    34  	}
    35  
    36  	/**
    37  	 * Get the name of the <code>Mediator</code>.
    38  	 * <P>
    39  	 * Override in subclass!</P>
    40  	 */		
    41  	public function getMediatorName() : String 
    42  	{	
    43  		return mediatorName;
    44  	}
    45  
    46  	
    47  	/**
    48  		 * Set the <code>IMediator</code>'s view component.
    49  		 * 
    50  		 * @param Object the view component
    51  		 */
    52  	public function setViewComponent( viewComponent:Object ):Void 
    53  	{
    54  		this.viewComponent = viewComponent;
    55  	}
    56  		
    57  	/**
    58  	 * Get the <code>Mediator</code>'s view component.
    59  	 * 
    60  	 * <P>
    61  	 * Additionally, an implicit getter will usually
    62  	 * be defined in the subclass that casts the view 
    63  	 * object to a type, like this:</P>
    64  	 * 
    65  	 * <listing>
    66  	 *		private function get comboBox : mx.controls.ComboBox 
    67  	 *		{
    68  	 *			return viewComponent as mx.controls.ComboBox;
    69  	 *		}
    70  	 * </listing>
    71  	 */		
    72  	public function getViewComponent() : Object
    73  	{	
    74  		return viewComponent;
    75  	}
    76  
    77  	/**
    78  	 * List the <code>INotification</code> names this
    79  	 * <code>Mediator</code> is interested in being notified of.
    80  	 * 
    81  	 * @return Array the list of <code>INotification</code> names 
    82  	 */
    83  	public function listNotificationInterests() : Array 
    84  	{
    85  		return [ ];
    86  	}
    87  
    88  	/**
    89  	 * Handle <code>INotification</code>s.
    90  	 * 
    91  	 * <P>
    92  	 * Typically this will be handled in a switch statement,
    93  	 * with one 'case' entry per <code>INotification</code>
    94  	 * the <code>Mediator</code> is interested in.
    95  	 */ 
    96  	public function handleNotification( notification:INotification ) : Void {}
    97  	
    98  	/**
    99  	 * Called by the View when the Mediator is registered
   100  	 */ 
   101  	public function onRegister( ) : Void {}
   102  
   103  	/**
   104  	 * Called by the View when the Mediator is removed
   105  	 */ 
   106  	public function onRemove( ) : Void {}
   107  	
   108  	// the mediator name
   109  	private var mediatorName:String;
   110  		
   111  	// The view component
   112  	private var viewComponent : Object;
   113  	
   114  }
