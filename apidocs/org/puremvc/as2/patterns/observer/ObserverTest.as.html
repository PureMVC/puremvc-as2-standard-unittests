    30  	/**
    31  	 * Tests observer class when initialized by accessor methods.
    32  	 * 
    33  	 */
    34  	public function testObserverAccessors():Void {
    35  		
    36  		// Create observer with null args, then
    37  		// use accessors to set notification method and context
    38  		var observer:Observer = new Observer(null, null);
    39  		observer.setNotifyContext(this);
    40  		observer.setNotifyMethod(observerTestMethod);
    41  		
    42  		// create a test event, setting a payload value and notify 
    43  		// the observer with it. since the observer is this class 
    44  		// and the notification method is observerTestMethod,
    45  		// successful notification will result in our local 
    46  		// observerTestVar being set to the value we pass in 
    47  		// on the note body.
    48  		var note:Notification = new Notification("ObserverTestNote", 10);
    49  		observer.notifyObserver(note);
    50  
    51  		// test assertions  			
    52  		assertTrue( "Expecting observerTestVar = 10", observerTestVar == 10 );
    53  	}
    54

    59  	public function testObserverConstructor():Void {
    60  		
    61  		// Create observer passing in notification method and context
    62  		var observer:Observer = new Observer(observerTestMethod,this);
    63  		
    64  		// create a test note, setting a body value and notify 
    65  		// the observer with it. since the observer is this class 
    66  		// and the notification method is observerTestMethod,
