     6  import org.puremvc.as2.interfaces.*;
     7  
     8  /**
     9  	 * The interface definition for a PureMVC Model.
    10  	 * 
    11  	 * <P>
    12  	 * In PureMVC, <code>IModel</code> implementors provide
    13  	 * access to <code>IProxy</code> objects by named lookup. </P>
    14  	 * 
    15  	 * <P>
    16  	 * An <code>IModel</code> assumes these responsibilities:</P>
    17  	 * 
    18  	 * <UL>
    19  	 * <LI>Maintain a cache of <code>IProxy</code> instances</LI>
    20  	 * <LI>Provide methods for registering, retrieving, and removing <code>IProxy</code> instances</LI>
    21  	 * </UL>
    22  	 */
    23  interface org.puremvc.as2.interfaces.IModel
    24  {
    25  	
    26  	/**
    27  	 * Register an <code>IProxy</code> instance with the <code>Model</code>.
    28  	 * 
    29  	 * @param proxyName the name to associate with this <code>IProxy</code> instance.
    30  	 * @param proxy an object reference to be held by the <code>Model</code>.
    31  	 */
    32  	function registerProxy( proxy:IProxy ) : Void;
    33  
    34  	/**
    35  	 * Retrieve an <code>IProxy</code> instance from the Model.
    36  	 * 
    37  	 * @param proxyName
    38  	 * @return the <code>IProxy</code> instance previously registered with the given <code>proxyName</code>.
    39  	 */
    40  	function retrieveProxy( proxyName:String ) : IProxy;
    41  
    42  	/**
    43  	 * Remove an <code>IProxy</code> instance from the Model.
    44  	 * 
    45  	 * @param proxyName name of the <code>IProxy</code> instance to be removed.
    46  	 */
    47  	function removeProxy( proxyName:String ) : IProxy;
    48  	
    49  	
    50  	/**
    51  	 * Check if a Proxy is registered
    52  	 * 
    53  	 * @param proxyName
    54  	 * @return whether a Proxy is currently registered with the given <code>proxyName</code>.
    55  	 */
    56  	function hasProxy( proxyName:String ) : Boolean;
    57  }
