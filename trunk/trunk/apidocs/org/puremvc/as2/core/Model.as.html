     6  import org.puremvc.as2.interfaces.*;
     7  
     8  /**
     9   * A Singleton <code>IModel</code> implementation.
    10   * 
    11   * <P>
    12   * In PureMVC, the <code>Model</code> class provides
    13   * access to model objects (Proxies) by named lookup. 
    14   * 
    15   * <P>
    16   * The <code>Model</code> assumes these responsibilities:</P>
    17   * 
    18   * <UL>
    19   * <LI>Maintain a cache of <code>IProxy</code> instances.</LI>
    20   * <LI>Provide methods for registering, retrieving, and removing 
    21   * <code>IProxy</code> instances.</LI>
    22   * </UL>
    23   * 
    24   * <P>
    25   * Your application must register <code>IProxy</code> instances 
    26   * with the <code>Model</code>. Typically, you use an 
    27   * <code>ICommand</code> to create and register <code>IProxy</code> 
    28   * instances once the <code>Facade</code> has initialized the Core 
    29   * actors.</p>
    30   *
    31   * @see org.puremvc.as2.patterns.proxy.Proxy Proxy
    32   * @see org.puremvc.as2.interfaces.IProxy IProxy
    33   */
    34  class org.puremvc.as2.core.Model implements IModel
    35  {
    36  
    37      /**
    38       * Constructor. 
    39       * 
    40       * <P>
    41       * This <code>IModel</code> implementation is a Singleton, 
    42       * so you should not call the constructor 
    43       * directly, but instead call the static Singleton 
    44       * Factory method <code>Model.getInstance()</code>
    45       * 
    46       * @throws Error Error if Singleton instance has already been constructed
    47       * 
    48       */
    49      public function Model( ) {
    50          if (instance != null) throw Error(SINGLETON_MSG);
    51          instance = this;
    52          proxyMap = new Array();	
    53          initializeModel();	
    54      }
    55  
    56      
    57      /**
    58       * Initialize the Singleton <code>Model</code> instance.
    59       * 
    60       * <P>
    61       * Called automatically by the constructor, this
    62       * is your opportunity to initialize the Singleton
    63       * instance in your subclass without overriding the
    64       * constructor.</P>
    65       * 
    66       * @return Void
    67       */
    68      private function initializeModel(  ): Void {
    69      }
    70  
    71      
    72      /**
    73       * <code>Model</code> Singleton Factory method.
    74       * 
    75       * @return the Singleton instance
    76       */
    77      public static function getInstance(): IModel {
    78          if (instance == null) instance = new Model();
    79          return instance;
    80      }
    81  
    82      
    83      /**
    84       * Register an <code>IProxy</code> with the <code>Model</code>.
    85       * 
    86       * @param proxy an <code>IProxy</code> to be held by the <code>Model</code>.
    87       */
    88      public function registerProxy( proxy: IProxy ): Void {
    89          proxyMap[ proxy.getProxyName() ] = proxy;
    90          proxy.onRegister();
    91      }
    92  
    93      
    94      /**
    95       * Retrieve an <code>IProxy</code> from the <code>Model</code>.
    96       * 
    97       * @param proxyName
    98       * @return the <code>IProxy</code> instance previously registered with the given <code>proxyName</code>.
    99       */
   100      public function retrieveProxy( proxyName: String ): IProxy {
   101          return proxyMap[ proxyName ];
   102      }
   103  
   104      
   105      /**
   106       * Check if a Proxy is registered
   107       * 
   108       * @param proxyName
   109       * @return whether a Proxy is currently registered with the given <code>proxyName</code>.
   110       */
   111      public function hasProxy( proxyName: String ): Boolean {
   112          return proxyMap[ proxyName ] != null;
   113      }
   114  
   115      
   116      /**
   117       * Remove an <code>IProxy</code> from the <code>Model</code>.
   118       * 
   119       * @param proxyName name of the <code>IProxy</code> instance to be removed.
   120       * @return the <code>IProxy</code> that was removed from the <code>Model</code>
   121       */
   122      public function removeProxy( proxyName: String ): IProxy {
   123          var proxy: IProxy = IProxy(proxyMap[ proxyName ]);
   124          proxyMap[ proxyName ] = null;
   125  		
   126          // alert the proxy that it has been removed
   127          if (proxy) proxy.onRemove();
   128          return proxy;
   129      }
   130  
   131      // Mapping of proxyNames to IProxy instances
   132      private var proxyMap: Array;
   133  
   134      // Singleton instance
   135      private static var instance: IModel;
   136  
   137      // Message Constants
   138      private var SINGLETON_MSG: String = "Model Singleton already constructed!";
   139  }
