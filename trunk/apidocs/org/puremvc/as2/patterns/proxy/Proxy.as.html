     6  import org.puremvc.as2.interfaces.*;
     7  import org.puremvc.as2.patterns.observer.*;
     8  
     9  /**
    10   * A base <code>IProxy</code> implementation. 
    11   * 
    12   * <P>
    13   * In PureMVC, <code>Proxy</code> classes are used to manage parts of the 
    14   * application's data model. </P>
    15   * 
    16   * <P>
    17   * A <code>Proxy</code> might simply manage a reference to a local data object, 
    18   * in which case interacting with it might involve setting and 
    19   * getting of its data in synchronous fashion.</P>
    20   * 
    21   * <P>
    22   * <code>Proxy</code> classes are also used to encapsulate the application's 
    23   * interaction with remote services to save or retrieve data, in which case, 
    24   * we adopt an asyncronous idiom; setting data (or calling a method) on the 
    25   * <code>Proxy</code> and listening for a <code>Notification</code> to be sent 
    26   * when the <code>Proxy</code> has retrieved the data from the service. </P>
    27   * 
    28   * @see org.puremvc.as2.core.model.Model Model
    29   */
    30  class org.puremvc.as2.patterns.proxy.Proxy extends Notifier implements IProxy, INotifier
    31  {
    32  
    33      public static var NAME: String = "Proxy";
    34  
    35      /**
    36       * Constructor
    37       */
    38      public function Proxy( proxyName: String, data: Object ) {
    39          this.proxyName = (proxyName != null) ? proxyName : NAME; 
    40          if (data != null) setData(data);
    41      }
    42  
    43      
    44      /**
    45       * Get the proxy name
    46       */
    47      public function getProxyName(): String {
    48          return proxyName;
    49      }		
    50  
    51      
    52      /**
    53       * Set the data object
    54       */
    55      public function setData( data: Object ): Void {
    56          this.data = data;
    57      }
    58  
    59      
    60      /**
    61       * Get the data object
    62       */
    63      public function getData(): Object {
    64          return data;
    65      }		
    66  
    67      
    68      /**
    69       * Called by the Model when the Proxy is registered
    70       */
    71      public function onRegister(): Void { 
    72      }		
    73  
    74      
    75      /**
    76       * Called by the Model when the Proxy is removed
    77       */
    78      public function onRemove(): Void { 
    79      }		
    80  
    81      
    82      // the proxy name
    83      private var proxyName: String;
    84  
    85      // the data object
    86      private var data: Object;
    87  }
