     6  import org.puremvc.as2.interfaces.*;
     7  import org.puremvc.as2.patterns.observer.Observer;
     8  
     9  /**
    10   * A Singleton <code>IView</code> implementation.
    11   * 
    12   * <P>
    13   * In PureMVC, the <code>View</code> class assumes these responsibilities:
    14   * <UL>
    15   * <LI>Maintain a cache of <code>IMediator</code> instances.</LI>
    16   * <LI>Provide methods for registering, retrieving, and removing <code>IMediators</code>.</LI>
    17   * <LI>Managing the observer lists for each <code>INotification</code> in the application.</LI>
    18   * <LI>Providing a method for attaching <code>IObservers</code> to an <code>INotification</code>'s observer list.</LI>
    19   * <LI>Providing a method for broadcasting an <code>INotification</code>.</LI>
    20   * <LI>Notifying the <code>IObservers</code> of a given <code>INotification</code> when it broadcast.</LI>
    21   * </UL>
    22   * 
    23   * @see org.puremvc.as2.patterns.mediator.Mediator Mediator
    24   * @see org.puremvc.as2.patterns.observer.Observer Observer
    25   * @see org.puremvc.as2.patterns.observer.Notification Notification
    26   */
    27  class org.puremvc.as2.core.View implements IView
    28  {
    29  
    30      public var name: String = "VIEW";
    31  
    32      /**
    33       * Constructor. 
    34       * 
    35       * <P>
    36       * This <code>IView</code> implementation is a Singleton, 
    37       * so you should not call the constructor 
    38       * directly, but instead call the static Singleton 
    39       * Factory method <code>View.getInstance()</code>
    40       * 
    41       * @throws Error Error if Singleton instance has already been constructed
    42       * 
    43       */
    44      public function View( ) {
    45          if (instance != null) throw Error(SINGLETON_MSG);
    46          instance = this;
    47          mediatorMap = new Array();
    48          observerMap = new Array();	
    49          initializeView();	
    50      }
    51  
    52      
    53      /**
    54       * Initialize the Singleton View instance.
    55       * 
    56       * <P>
    57       * Called automatically by the constructor, this
    58       * is your opportunity to initialize the Singleton
    59       * instance in your subclass without overriding the
    60       * constructor.</P>
    61       * 
    62       * @return Void
    63       */
    64      private function initializeView(  ): Void {
    65      }
    66  
    67      
    68      /**
    69       * View Singleton Factory method.
    70       * 
    71       * @return the Singleton instance of <code>View</code>
    72       */
    73      public static function getInstance(): IView {
    74          if ( instance == null ) instance = new View();
    75          return instance;
    76      }
    77  
    78      
    79      /**
    80       * Register an <code>IObserver</code> to be notified
    81       * of <code>INotifications</code> with a given name.
    82       * 
    83       * @param notificationName the name of the <code>INotifications</code> to notify this <code>IObserver</code> of
    84       * @param observer the <code>IObserver</code> to register
    85       */
    86      public function registerObserver( notificationName: String, observer: IObserver ): Void {
    87          var observers: Array = observerMap[ notificationName ];
    88          if( observers ) {
    89              observers.push(observer);
    90          } else {
    91              observerMap[ notificationName ] = [observer];	
    92          }
    93      }
    94  
    95      
    96      /**
    97       * Notify the <code>IObservers</code> for a particular <code>INotification</code>.
    98       * 
    99       * <P>
   100       * All previously attached <code>IObservers</code> for this <code>INotification</code>'s
   101       * list are notified and are passed a reference to the <code>INotification</code> in 
   102       * the order in which they were registered.</P>
   103       * 
   104       * @param notification the <code>INotification</code> to notify <code>IObservers</code> of.
   105       */
   106      public function notifyObservers( notification: INotification ): Void {
   107          if( observerMap[ notification.getName() ] != undefined && observerMap[ notification.getName() ] != null) {
   108              //[PB 01] Should cast observer map to Array, but can't (and don't actually need to)
   109              var observers: Array = observerMap[ notification.getName()];
   110              for (var i: Number = 0;i < observers.length; i++) {
   111                  var observer: IObserver = IObserver(observers[ i ]);
   112                  observer.notifyObserver(notification);
   113              }
   114          }
   115      }
   116  
   117      
   118      /**
   119       * Remove the observer for a given notifyContext from an observer list for a given Notification name.
   120       * <P>
   121       * @param notificationName which observer list to remove from 
   122       * @param notifyContext remove the observer with this object as its notifyContext
   123       */
   124      public function removeObserver( notificationName: String, notifyContext: Object ): Void {
   125          // the observer list for the notification under inspection
   126          var observers: Array = observerMap[ notificationName ];
   127  
   128          // find the observer for the notifyContext
   129          for ( var i: Number = 0;i < observers.length; i++ ) {
   130              if ( Observer(observers[i]).compareNotifyContext(notifyContext) == true ) {
   131                  // there can only be one Observer for a given notifyContext 
   132                  // in any given Observer list, so remove it and break
   133                  observers.splice(i, 1);
   134                  break;
   135              }
   136          }
   137  
   138          // Also, when a Notification's Observer list length falls to 
   139          // zero, delete the notification key from the observer map
   140          if ( observers.length == 0 ) {
   141              delete observerMap[ notificationName ];		
   142          }
   143      } 
   144  
   145      
   146      /**
   147       * Register an <code>IMediator</code> instance with the <code>View</code>.
   148       * 
   149       * <P>
   150       * Registers the <code>IMediator</code> so that it can be retrieved by name,
   151       * and further interrogates the <code>IMediator</code> for its 
   152       * <code>INotification</code> interests.</P>
   153       * <P>
   154       * If the <code>IMediator</code> returns any <code>INotification</code> 
   155       * names to be notified about, an <code>Observer</code> is created encapsulating 
   156       * the <code>IMediator</code> instance's <code>handleNotification</code> method 
   157       * and registering it as an <code>Observer</code> for all <code>INotifications</code> the 
   158       * <code>IMediator</code> is interested in.</p>
   159       * 
   160       * @param mediatorName the name to associate with this <code>IMediator</code> instance
   161       * @param mediator a reference to the <code>IMediator</code> instance
   162       */
   163      public function registerMediator( mediator: IMediator ): Void {
   164          // Register the Mediator for retrieval by name
   165          mediatorMap[ mediator.getMediatorName() ] = mediator;
   166          // Get Notification interests, if any.
   167          var interests: Array = mediator.listNotificationInterests();
   168  		
   169          // Register Mediator as an observer for each notification of interests
   170          if ( interests.length > 0) {
   171              // Create Observer referencing this mediator's handlNotification method
   172              var observer: Observer = new Observer(mediator.handleNotification, mediator);
   173  
   174              // Register Mediator as Observer for its list of Notification interests
   175              for ( var i: Number = 0;i < interests.length; i++ ) {
   176                  registerObserver(interests[i], observer);
   177              }			
   178          }
   179  			
   180          // alert the mediator that it has been registered
   181          mediator.onRegister();
   182      }
   183  
   184      
   185      /**
   186       * Retrieve an <code>IMediator</code> from the <code>View</code>.
   187       * 
   188       * @param mediatorName the name of the <code>IMediator</code> instance to retrieve.
   189       * @return the <code>IMediator</code> instance previously registered with the given <code>mediatorName</code>.
   190       */
   191      public function retrieveMediator( mediatorName: String ): IMediator {
   192          return mediatorMap[ mediatorName ];
   193      }
   194  
   195      
   196      /**
   197       * Remove an <code>IMediator</code> from the <code>View</code>.
   198       * 
   199       * @param mediatorName name of the <code>IMediator</code> instance to be removed.
   200       * @return the <code>IMediator</code> that was removed from the <code>View</code>
   201       */
   202      public function removeMediator( mediatorName: String ): IMediator {
   203          // Retrieve the named mediator
   204          var mediator: IMediator = IMediator(mediatorMap[ mediatorName ]) ;
   205  			
   206          if ( mediator ) {
   207              // for every notification this mediator is interested in...
   208              var interests: Array = mediator.listNotificationInterests();
   209              for ( var i: Number = 0;i < interests.length; i++ ) {
   210                  // remove the observer linking the mediator 
   211                  // to the notification interest
   212                  removeObserver(interests[i], mediator);
   213              }	
   214  				
   215              // remove the mediator from the map		
   216              delete mediatorMap[ mediatorName ];
   217  	
   218              // alert the mediator that it has been removed
   219              mediator.onRemove();
   220          }
   221  			
   222          return mediator;
   223      }
   224  
   225      
   226      /**
   227       * Check if a Mediator is registered or not
   228       * 
   229       * @param mediatorName
   230       * @return whether a Mediator is registered with the given <code>mediatorName</code>.
   231       */
   232      public function hasMediator( mediatorName: String ): Boolean {
   233          return mediatorMap[ mediatorName ] != null;
   234      }
   235  
   236      // Mapping of Mediator names to Mediator instances
   237      private var mediatorMap: Array;
   238  
   239      // Mapping of Notification names to Observer lists
   240      private var observerMap: Array;
   241  
   242      // Singleton instance
   243      private static var instance: IView;
   244  
   245      // Message Constants
   246      private var SINGLETON_MSG: String = "View Singleton already constructed!";
   247  }
